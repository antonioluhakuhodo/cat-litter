<script>
    // 初始化資料
    let dataLog = JSON.parse(localStorage.getItem('catLitterLog')) || [];

    // 格式化日期時間 (YYYY/MM/DD HH:mm)
    function formatDateTime() {
        const now = new Date();
        const yyyy = now.getFullYear();
        const mm = String(now.getMonth() + 1).padStart(2, '0');
        const dd = String(now.getDate()).padStart(2, '0');
        const hh = String(now.getHours()).padStart(2, '0');
        const min = String(now.getMinutes()).padStart(2, '0');
        return `${yyyy}/${mm}/${dd} ${hh}:${min}`;
    }

    // 新增紀錄
    function addLog(name) {
        const timestamp = formatDateTime();
        const id = Date.now(); // 用時間戳當唯一 ID 方便刪除
        
        dataLog.unshift({ id, name, time: timestamp }); // 新的排前面
        saveAndRender();
    }

    // 刪除單筆紀錄
    function deleteLog(id) {
        if(confirm('確定要刪除這條紀錄嗎？')) {
            dataLog = dataLog.filter(item => item.id !== id);
            saveAndRender();
        }
    }

    // 清除全部資料
    function clearAll() {
        if (confirm('確定要清空所有紀錄嗎？此操作無法復原！')) {
            dataLog = [];
            saveAndRender();
        }
    }

    // 儲存並更新畫面
    function saveAndRender() {
        localStorage.setItem('catLitterLog', JSON.stringify(dataLog));
        render();
    }

    // 渲染畫面與統計
    function render() {
        const listEl = document.getElementById('logList');
        listEl.innerHTML = '';

        let countA = 0;
        let countB = 0;

        dataLog.forEach(item => {
            // 計算統計
            if (item.name === '室友 A') countA++;
            if (item.name === '室友 B') countB++;

            // 產生列表 HTML
            const li = document.createElement('li');
            li.className = 'log-item';
            li.innerHTML = `
                <span><strong>${item.name}</strong> <small style="color:#666; margin-left:8px;">${item.time}</small></span>
                <button class="delete-btn" onclick="deleteLog(${item.id})">❌</button>
            `;
            listEl.appendChild(li);
        });

        // 更新統計文字
        document.getElementById('stat-user1').innerText = `室友 A: ${countA} 次`;
        document.getElementById('stat-user2').innerText = `室友 B: ${countB} 次`;
    }

    // 網頁載入時先渲染一次
    render();
</script>
